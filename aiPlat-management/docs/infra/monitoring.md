# 监控告警

> 指标监控、告警规则、审计日志管理

---

## 一、模块定位

监控告警负责基础设施层的指标采集、告警管理、审计日志等监控运维功能。

### 核心职责

```
监控告警
├── 指标监控
│   ├── 系统指标（CPU/内存/磁盘/网络）
│   ├── GPU 指标（利用率/显存/温度/功耗）
│   ├── 服务指标（QPS/延迟/错误率）
│   └── 自定义指标
│
├── 告警规则
│   ├── 告警规则配置
│   ├── 告警阈值管理
│   ├── 告警通知渠道
│   └── 告警历史查询
│
└── 审计日志
    ├── 操作日志
    ├── 系统日志
    ├── 安全日志
    └── 日志查询分析
```

### 指标监控

```python
# 获取集群概览
GET /api/infra/monitoring/overview

# 获取节点指标
GET /api/infra/monitoring/nodes/{node_name}/metrics

# 获取服务指标
GET /api/infra/monitoring/services/{service_name}/metrics
Query: start, end, step

# 获取 GPU 指标
GET /api/infra/monitoring/gpu/metrics
Query: node, gpu_id, start, end

# 自定义查询
POST /api/infra/monitoring/query
Body: {
  "query": "avg(rate(container_cpu_usage_seconds_total[5m]))",
  "start": "2024-01-15T00:00:00Z",
  "end": "2024-01-15T23:59:59Z",
  "step": "1m"

# 获取仪表盘配置
GET /api/infra/monitoring/dashboards/{dashboard_id}

Body: {
  "name": "custom-dashboard",
  "panels": [...]
}
```

### 告警规则

```python
# 获取告警规则列表
GET /api/infra/alerting/rules
Query: type, severity, status

# 创建告警规则
POST /api/infra/alerting/rules
Body: {
  "name": "gpu-overload",
  "type": "gpu",
  "metric": "gpu_utilization",
  "condition": ">",
  "threshold": 95,
  "duration": 300,
  "severity": "critical",
  "labels": {"node_type": "gpu"},
  "notifications": [
    {"type": "email", "target": "admin@example.com"},
    {"type": "webhook", "target": "https://..."}
  ]
}

# 更新告警规则
PUT /api/infra/alerting/rules/{rule_id}

# 删除告警规则
DELETE /api/infra/alerting/rules/{rule_id}

# 启用/停用规则
POST /api/infra/alerting/rules/{rule_id}/toggle
Body: {"enabled": True}

# 测试触发
POST /api/infra/alerting/rules/{rule_id}/test
```

## 四、数据模型

```typescript
// 指标数据
interface Metric {
  name: string;
  value: number;
  unit: string;
  timestamp: string;
  labels: Record<string, string>;
}

// 告警规则
interface AlertRule {
  id: string;
  name: string;
  type: 'system' | 'gpu' | 'storage' | 'service';
  metric: string;
  condition: '>' | '<' | '==' | '!=' | '>=' | '<=';
  threshold: number;
  duration: number;
  severity: 'critical' | 'warning' | 'info';
  status: 'enabled' | 'disabled';
  labels: Record<string, string>;
  notifications: Notification[];
  createdAt: string;
  updatedAt: string;
}

// 通知配置
interface Notification {
  type: 'email' | 'dingtalk' | 'webhook' | 'slack';
  target: string;
  template?: string;
}

// 告警历史
interface AlertHistory {
  id: string;
  ruleId: string;
  ruleName: string;
  severity: 'critical' | 'warning' | 'info';
  status: 'firing' | 'resolved' | 'acknowledged' | 'silenced';
  labels: Record<string, string>;
  value: number;
  duration: number;
  notifications: NotificationRecord[];
  firedAt: string;
  resolvedAt?: string;
}

// 通知记录
interface NotificationRecord {
  timestamp: string;
  channel: string;
  status: 'success' | 'failed';
  recipient: string;
}

// 审计日志
interface AuditLog {
  id: string;
  timestamp: string;
  user: string;
  operation: 'create' | 'update' | 'delete' | 'login' | 'logout';
  resourceType: string;
  resourceName: string;
  result: 'success' | 'failed';
  changes?: {
    before: any;
    after: any;
  };
  clientIp: string;
  userAgent: string;
  requestId: string;
}

// 仪表盘
interface Dashboard {
  id: string;
  name: string;
  panels: DashboardPanel[];
  createdAt: string;
  updatedAt: string;
}

// 仪表盘面板
interface DashboardPanel {
  id: string;
  title: string;
  type: 'graph' | 'gauge' | 'table' | 'text';
  metrics: string[];
  refresh: number;
  position: {
    x: number;
    y: number;
    width: number;
    height: number;
  };
}
```
