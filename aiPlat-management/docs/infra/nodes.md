# 节点管理

> GPU 物理节点或 K8s 工作节点的完整生命周期管理

---

## 一、模块定位

节点管理负责 GPU 物理节点或 K8s 工作节点的添加、删除、监控和维护。

### 核心职责

```
节点管理
├── 节点生命周期
│   ├── 添加节点（云厂商/物理机）
│   ├── 删除节点（驱逐Pod/释放资源）
│   ├── 节点扩缩容
│   └── 节点维护（标签/污点）
│
├── 节点监控
│   ├── GPU使用率（利用率/显存/温度）
│   ├── 节点健康状态
│   ├── 资源使用（CPU/内存/磁盘）
│   └── 网络状态
│
└── 驱动管理
    ├── 驱动版本检查
    ├── 驱动升级
    └── 驱动回滚
```

## 三、API 端点

### 节点管理

```python
# 获取节点列表
GET /api/infra/nodes
Response: {
  "nodes": [
    {
      "name": "node-01",
      "ip": "10.0.0.101",
      "gpu_model": "A100",
      "gpu_count": 4,
      "driver_version": "535.54.03",
      "status": "Ready",
      "gpus": [...]
    }
  ],
  "total": 8,
  "healthy": 7
}

# 获取节点详情
GET /api/infra/nodes/{node_name}

# 添加节点
POST /api/infra/nodes
Body: {
  "provider": "aws",
  "instanceType": "p4d.24xlarge",
  "count": 2,
  "labels": {"gpu": "true"}
}

# 删除节点
DELETE /api/infra/nodes/{node_name}

# 驱逐节点
POST /api/infra/nodes/{node_name}/drain

# 重启节点
POST /api/infra/nodes/{node_name}/restart

# 更新节点标签
PUT /api/infra/nodes/{node_name}/labels
```

### 驱动管理

```python
# 获取驱动列表
GET /api/infra/drivers

# 升级驱动
POST /api/infra/drivers/upgrade
Body: {
  "version": "535.104.05",
  "nodes": ["node-01", "node-02"]
}

# 回滚驱动
POST /api/infra/drivers/rollback
Body: {
  "version": "525.60.13",
  "nodes": ["node-01"]
}
```

## 四、数据模型

```typescript
// 节点
interface Node {
  name: string;              // 节点名称
  ip: string;                // IP地址
  gpu_model: string;         // GPU型号
  gpu_count: number;         // GPU数量
  driver_version: string;    // 驱动版本
  status: 'Ready' | 'NotReady' | 'Unknown';
  conditions: NodeCondition[];
  gpus: GPU[];
  labels: Record<string, string>;
  allocatable: Resource;
  capacity: Resource;
}

// GPU
interface GPU {
  id: string;                // GPU ID
  model: string;             // 型号
  utilization: number;       // 利用率
  memoryUsed: number;        // 已用显存
  memoryTotal: number;       // 总显存
  temperature: number;        // 温度
  powerUsage: number;        // 功耗
}

// 节点状态
interface NodeCondition {
  type: string;              // 状态类型
  status: string;            // 状态值
  message: string;           // 消息
  lastTransitionTime: string;
}
```
