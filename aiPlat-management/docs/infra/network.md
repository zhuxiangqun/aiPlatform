# 网络管理

> 服务发现、负载均衡、网络策略管理

---

## 一、模块定位

网络管理负责服务发现、负载均衡、网络策略等网络相关功能的完整管理。

### 核心职责

```
网络管理
├── 服务发现
│   ├── Kubernetes Service 管理
│   ├── Endpoints 管理
│   ├── DNS 配置管理
│   └── 服务注册发现
│
├── 负载均衡
│   ├── LoadBalancer 管理
│   ├── Ingress 管理
│   ├── 流量分配策略
│   └── 健康检查配置
│
└── 网络策略
    ├── NetworkPolicy 管理
    ├── 流量控制
    ├── 安全隔离
    └── 网络排查
```

## 三、API 端点

### 服务发现

```python
# 获取服务列表
GET /api/infra/network/services
Query: namespace, type

# 创建服务
POST /api/infra/network/services
Body: {
  "name": "gpt4-api",
  "namespace": "ai-prod",
  "type": "ClusterIP",
  "ports": [
    {"name": "http", "port": 8080, "targetPort": 8080, "protocol": "TCP"}
  ],
  "selector": {"app": "gpt4-api"}
}

# 更新服务
PUT /api/infra/network/services/{service_name}

# 删除服务
DELETE /api/infra/network/services/{service_name}

# 获取 Endpoints
GET /api/infra/network/services/{service_name}/endpoints

# DNS 解析测试
POST /api/infra/network/dns/resolve
Body: {
  "hostname": "gpt4-api.ai-prod.svc.cluster.local"
}
```

### 负载均衡

```python
# 获取 Ingress 列表
GET /api/infra/network/ingress
Query: namespace

# 创建 Ingress
POST /api/infra/network/ingress
Body: {
  "name": "api-ingress",
  "namespace": "ai-prod",
  "host": "api.ai.com",
  "tls": True,
  "rules": [
    {
      "path": "/",
      "backend": {"serviceName": "gpt4-api", "servicePort": 8080}
    }
  ]
}

# 更新 Ingress
PUT /api/infra/network/ingress/{ingress_name}

# 删除 Ingress
DELETE /api/infra/network/ingress/{ingress_name}

# 获取 TLS 证书
GET /api/infra/network/ingress/{ingress_name}/tls

# 配置健康检查
POST /api/infra/network/ingress/{ingress_name}/healthcheck
Body: {
  "path": "/health",
  "interval": "10s",
  "timeout": "5s",
  "failureThreshold": 3
}
```

## 四、数据模型

```typescript
// Service
interface Service {
  name: string;
  namespace: string;
  type: 'ClusterIP' | 'NodePort' | 'LoadBalancer';
  clusterIP: string;
  ports: ServicePort[];
  selector: Record<string, string>;
  endpoints: Endpoint[];
  createdAt: string;
}

// ServicePort
interface ServicePort {
  name: string;
  port: number;
  targetPort: number;
  protocol: 'TCP' | 'UDP';
  nodePort?: number;
}

// Endpoint
interface Endpoint {
  pod: string;
  ip: string;
  port: number;
  ready: boolean;
}

// Ingress
interface Ingress {
  name: string;
  namespace: string;
  className: string;
  host: string;
  tls: boolean;
  tlsSecret?: string;
  rules: IngressRule[];
  address: string;
  createdAt: string;
}

// IngressRule
interface IngressRule {
  host: string;
  path: string;
  backend: {
    serviceName: string;
    servicePort: number;
  };
}

// NetworkPolicy
interface NetworkPolicy {
  name: string;
  namespace: string;
  type: 'Ingress' | 'Egress' | 'Both';
  podSelector: Record<string, string>;
  ingress?: PolicyRule[];
  egress?: PolicyRule[];
  status: 'enabled' | 'disabled';
  createdAt: string;
}

// PolicyRule
interface PolicyRule {
  from?: NetworkPeer[];
  to?: NetworkPeer[];
  ports: PolicyPort[];
}

// NetworkPeer
interface NetworkPeer {
  ipBlock?: string;
  namespaceSelector?: Record<string, string>;
  podSelector?: Record<string, string>;
}

// PolicyPort
interface PolicyPort {
  port: number;
  protocol: 'TCP' | 'UDP';
}
```
