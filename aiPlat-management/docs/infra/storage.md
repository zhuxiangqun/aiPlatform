# 存储管理

> 向量存储、模型存储、PVC 生命周期管理

---

## 一、模块定位

存储管理负责向量数据库、模型存储、持久化存储等资源的完整生命周期管理。

### 核心职责

```
存储管理
├── 向量存储
│   ├── 向量数据库管理（Qdrant/Milvus/Weaviate）
│   ├── Collection 管理
│   ├── 向量索引管理
│   └── 存储空间统计
│
├── 模型存储
│   ├── 模型仓库管理
│   ├── 模型版本管理
│   ├── 模型缓存策略
│   └── 存储空间统计
│
└── PVC 管理
    ├── PVC 创建/删除
    ├── 存储类管理
    ├── 存储配额管理
    └── 存储快照/恢复
```

## 三、API 端点

### 向量存储

```python
# 获取向量数据库状态
GET /api/infra/storage/vector/status

# 获取 Collection 列表
GET /api/infra/storage/vector/collections
Query: name, status

# 创建 Collection
POST /api/infra/storage/vector/collections
Body: {
  "name": "docs-emb",
  "dimension": 768,
  "indexType": "HNSW",
  "metric": "cosine",
  "config": {
    "M": 16,
    "efConstruct": 256
  }
}

# 删除 Collection
DELETE /api/infra/storage/vector/collections/{collection_name}

# 优化索引
POST /api/infra/storage/vector/collections/{collection_name}/optimize

# 导出向量
GET /api/infra/storage/vector/collections/{collection_name}/export
```

### 模型存储

```python
# 获取模型列表
GET /api/infra/storage/models
Query: type, cached

# 获取模型详情
GET /api/infra/storage/models/{model_id}

# 上传模型
POST /api/infra/storage/models/upload
Body: multipart/form-data

# 下载模型
GET /api/infra/storage/models/{model_id}/download

# 缓存模型到节点
POST /api/infra/storage/models/{model_id}/cache
Body: {
  "nodes": ["node-01", "node-02"]
}

# 清除缓存
POST /api/infra/storage/models/{model_id}/clear-cache

# 删除模型版本
DELETE /api/infra/storage/models/{model_id}/versions/{version}
```

### PVC 管理

```python
# 获取 PVC 列表
GET /api/infra/storage/pvc
Query: namespace, storageClass, status

# 创建 PVC
POST /api/infra/storage/pvc
Body: {
  "name": "model-cache",
  "namespace": "ai-prod",
  "storageClass": "fast-ssd",
  "size": "200Gi",
  "accessMode": "ReadWriteMany"
}

# 扩容 PVC
POST /api/infra/storage/pvc/{pvc_name}/resize
Body: {
  "size": "300Gi"
}

# 创建快照
POST /api/infra/storage/pvc/{pvc_name}/snapshot

# 恢复快照
POST /api/infra/storage/pvc/{pvc_name}/restore
Body: {
  "snapshotName": "snap-001"
}

# 删除 PVC
DELETE /api/infra/storage/pvc/{pvc_name}

# 获取存储类列表
GET /api/infra/storage/classes
```

## 四、数据模型

```typescript
// 向量 Collection
interface VectorCollection {
  name: string;
  vectorCount: number;
  dimension: number;
  indexType: 'HNSW' | 'IVF' | 'FLAT';
  metric: 'cosine' | 'euclidean' | 'dot_product';
  size: number;
  config: {
    M?: number;
    efConstruct?: number;
    efSearch?: number;
  };
  performance: {
    qps: number;
    avgLatency: number;
    p99Latency: number;
    memoryUsage: number;
  };
  createdAt: string;
  updatedAt: string;
}

// 模型
interface Model {
  id: string;
  name: string;
  type: 'LLM' | 'Embed' | 'Image' | 'Audio';
  latestVersion: string;
  size: number;
  cached: boolean;
  source: string;
  path: string;
  versions: ModelVersion[];
}

// 模型版本
interface ModelVersion {
  version: string;
  size: number;
  uploadedAt: string;
  cached: boolean;
  checksum: string;
}

// PVC
interface PVC {
  name: string;
  namespace: string;
  storageClass: string;
  size: string;
  usedSize: string;
  status: 'Bound' | 'Pending' | 'Released';
  accessMode: 'ReadWriteOnce' | 'ReadWriteMany' | 'ReadOnlyMany';
  reclaimPolicy: 'Retain' | 'Delete';
  mountPoints: MountPoint[];
  snapshots: Snapshot[];
  createdAt: string;
}

// 挂载点
interface MountPoint {
  pod: string;
  node: string;
  path: string;
  mode: 'RW' | 'RO';
}

// 快照
interface Snapshot {
  name: string;
  createdAt: string;
  size: string;
  status: 'Completed' | 'Creating' | 'Failed';
}
```
